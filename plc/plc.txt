PROGRAM PLC_PRG
VAR
	K0 AT %Q* : BOOL;
	APositive AT %Q* : BOOL;
	BPositive AT %Q* : BOOL;
	CPositive AT %Q* : BOOL;
	ANegative AT %Q* : BOOL;
	BNegative AT %Q* : BOOL;
	CNegative AT %Q* : BOOL;
	c0 AT %I* : BOOL;
	a1 AT %I* : BOOL;
	b1 AT %I* : BOOL;
	c1 AT %I* : BOOL;
	a0 AT %I* : BOOL;
	b0 AT %I* : BOOL;
	START : BOOL;
END_VAR

K0 := FALSE;
APositive := FALSE;
BPositive := FALSE;
CPositive := FALSE;
ANegative := FALSE;
BNegative := FALSE;
CNegative := FALSE;
c0 := TRUE;
a1 := FALSE;
b1 := FALSE;
c1 := FALSE;
a0 := TRUE;
b0 := TRUE;

WHILE START = True DO
IF START AND c0 AND NOT K0 THEN
	APositive := TRUE;
	IF a1 = True THEN
		BPositive := TRUE;
	END_IF;
	IF b1 = True THEN
		CPositive := TRUE;
	END_IF;
	IF c1 = True THEN
		K0 := TRUE;
	END_IF;

END_IF;


IF K0 = True THEN
	Apositive := FALSE;
	Bpositive := FALSE;
	Cpositive := FALSE;
END_IF;

IF c1 = True AND K0 = True THEN
	ANegative := TRUE;
	IF a0 = True THEN
		BNegative := TRUE;
	IF b0 = True THEN
		CNegative := TRUE;
	IF c0 = True THEN
		K0 := FALSE;
	END_IF;
END_IF;
IF K0 = False THEN
	Anegative := FALSE;
	Bnegative := FALSE;
	Cnegative := FALSE;
END_IF;

END_WHILE
